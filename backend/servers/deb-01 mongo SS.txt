apiVersion: apps/v1
kind: StatefulSet
metadata:
  name: mongodb
spec:
  serviceName: mongodb
  replicas: 1
  selector:
    matchLabels:
      app: mongodb
  template:
    metadata:
      labels:
        app: mongodb
    spec:
      securityContext:
        fsGroup: 999
        runAsUser: 999
        runAsGroup: 999
      tolerations:
      - key: "node.kubernetes.io/not-ready"
        operator: "Exists"
        effect: "NoSchedule"
      - key: "node-role.kubernetes.io/control-plane"
        operator: "Exists"
        effect: "NoSchedule"
      containers:
        - name: mongodb
          image: mongo:6.0
          ports:
            - containerPort: 27017
          resources:
            requests:
              memory: "1Gi"  # Увеличено с 1Gi
              cpu: "1"       # Увеличено с 500m
            limits:
              memory: "2Gi"  # Увеличено с 2Gi
              cpu: "2"       # Увеличено с 1
          livenessProbe:     # Добавлен liveness probe
            exec:
              command:
              - mongosh
              - --eval
              - "db.adminCommand('ping')"
            initialDelaySeconds: 30
            periodSeconds: 10
            timeoutSeconds: 5
            failureThreshold: 3
          readinessProbe:    # Добавлен readiness probe
            exec:
              command:
              - mongosh
              - --eval
              - "db.adminCommand('ping')"
            initialDelaySeconds: 5
            periodSeconds: 10
            timeoutSeconds: 5
          lifecycle:         # Добавлен lifecycle hook для graceful shutdown
            preStop:
              exec:
                command: ["mongosh", "--eval", "db.adminCommand('shutdown')"]
          volumeMounts:
            - name: mongodb01-data
              mountPath: /data/db
            - name: mongodb-config
              mountPath: /etc/mongod.conf
              subPath: mongod.conf
            - name: mongodb-logs   # Добавлен volume для логов
              mountPath: /var/log/mongodb
      volumes:
        - name: mongodb-config
          configMap:
            name: mongodb-config
        - name: mongodb-logs      # Добавлен emptyDir для логов
          emptyDir: {}
  volumeClaimTemplates:
  - metadata:
      name: mongodb01-data
    spec:
      accessModes: ["ReadWriteOnce"]
      resources:
        requests:
          storage: 8Gi
      storageClassName: manual
